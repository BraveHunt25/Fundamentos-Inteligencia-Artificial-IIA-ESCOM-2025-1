\section{Escribe las tablas de verdad para las operaciones not, and, or, or exclusivo, implicación y doble implicación}
\subsection{NOT}
\begin{table}[!ht]
    \centering
    \begin{tabular}{|c|c|}
        \hline
        a & $\overline{a}$\\
        \hline
        0 & 1 \\
        \hline
        1 & 0 \\
        \hline
    \end{tabular}
    \caption{Tabla de verdad: NOT}\label{table:not}
\end{table}

        \hline
