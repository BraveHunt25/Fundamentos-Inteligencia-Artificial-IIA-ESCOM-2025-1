\section{Usando solo el operador NAND $\overline{\wedge}$, construye las tablas de verdad del ejercicio 1}
\subsection{Para NOT}
\begin{table}[!ht]
    \centering
    \begin{tabular}{|c|c|}
        \hline
        a & \rule{0pt}{2.5ex}$a \overline{\wedge} a$ \\
        \hline
        0 & 1 \\
        \hline
        1 & 0 \\
        \hline
    \end{tabular}
    \caption{Tabla de verdad con NAND:NOT}\label{table:NAND-NOT}
\end{table}
